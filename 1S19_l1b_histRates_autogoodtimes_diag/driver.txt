input=./input
input_de=./input_de
output=./output

# Check if the target is a directory
if [ ! -d "$input" ]; then
  echo "Error: Directory '$directory' not found."
  exit 1
fi

for file in "$input"/*; do
  key=$(basename "$file" | grep -oE '_[0-9]{8}-repoint')

  match=$(ls "$input_de"/*"$key"* 2>/dev/null | head -n 1)

  if [[ -n "$match" ]]; then
    echo "MATCH:"
    echo "  hist = $file"
    echo "  de   = $match"

  # Check if the current item is a regular file (not a directory or other type)
    if [ -f "$file" ]; then
      echo "Processing l1b HIST file: $file"
    fi

    if [ -f "$match" ]; then
      echo "Processing l1b DE file: $match"
    fi

  else
    echo "NO MATCH for $file"
    match="$file"
  fi

  ofile="${file/input/output}"
  ofile="${ofile/.cdf/}"
  echo "Output GT file: $ofile"

  ./runIMAP-fmv1-autogt.sh "$file" "$match" "$ofile" 
  
done

cd $output 
cat imap_lo_goodtimes_*.csv > imap_lo_goodtimes.csv
cat imap_lo_H_background_*.csv > imap_lo_H_background.csv 
cat imap_lo_O_background_*.csv > imap_lo_O_background.csv 
cat imap_lo_HO_cnts_expo_*.csv > imap_lo_HO_cnts_expo.csv
cd ..
